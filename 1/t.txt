            СписокКандидатов = Новый Массив;
            
            // 1. Собираем список подходящих станков
            Если ЗначениеЗаполнено(ВыборкаОпераций.ТребуемыйРЦ) Тогда
                // Если жестко задан станок - он единственный кандидат
                СписокКандидатов.Добавить(ВыборкаОпераций.ТребуемыйРЦ);
            Иначе
                // Ищем ВСЕ станки такого типа
                ЗапросРЦ = Новый Запрос("ВЫБРАТЬ Ссылка ИЗ Справочник.РабочиеЦентры ГДЕ ВидОперации = &Вид");
                ЗапросРЦ.УстановитьПараметр("Вид", ВыборкаОпераций.ВидОперации);
                ВыборкаРЦ = ЗапросРЦ.Выполнить().Выбрать();
                Пока ВыборкаРЦ.Следующий() Цикл
                    СписокКандидатов.Добавить(ВыборкаРЦ.Ссылка);
                КонецЦикла;
            КонецЕсли;
            
            // 2. Выбираем ЛУЧШИЙ станок из кандидатов
            ЛучшийРЦ = Неопределено;
            ЛучшийИнтервал = Неопределено;
            МаксимальноеНачало = Дата(1,1,1); // Самая ранняя дата
            
            Для Каждого КандидатРЦ Из СписокКандидатов Цикл
                
                // Пробуем запланировать на этом станке
                ВозможныйИнтервал = НайтиСвободноеОкно(КандидатРЦ, ВремяНаПартию, ТекущийДедлайн, ТаблицаЗанятости, СтрокаЗаказа.Цвет);
                
                // Сравниваем: этот вариант лучше предыдущего?
                // Стратегия JIT: Чем ПОЗЖЕ мы начнем (ближе к дедлайну), тем лучше.
                Если ВозможныйИнтервал.Начало > МаксимальноеНачало Тогда
                    МаксимальноеНачало = ВозможныйИнтервал.Начало;
                    ЛучшийИнтервал = ВозможныйИнтервал;
                    ЛучшийРЦ = КандидатРЦ;
                КонецЕсли;
                
            КонецЦикла;
